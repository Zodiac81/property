2) Use

	docker [опция] [команда] [аргументы]
	docker

	docker search ubuntu
	docker pull ubuntu
	docker run ubuntu
	docker images 					// see images

	docker run -it ubuntu 			// root@d9b100f2f636:/#
		apt-get update
		exit

	docker ps 
				[-a] 				// all conteiners
				[-a] 				// last conteiner

	docker stop < ID or repository_name > 

	docker rmi [-f] < ID or repository_name > 

	docker exec -ti < repository_name >  bash

	docker container ls

docker exec  composer install

Kill all running containers
	docker kill $(docker ps -q)
	docker rm $(docker ps -a -q)
	docker rmi $(docker images -q -f dangling=true)
	docker rmi $(docker images -q)
